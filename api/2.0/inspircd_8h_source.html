122 template<typename T> inline std::string ConvNumeric(const T &in)
123 {
124  if (in == 0) return "0";
125  char res[MAXBUF];
126  char* out = res;
127  T quotient = in;
128  while (quotient) {
129  *out = "0123456789"[ std::abs( (long)quotient % 10 ) ];
130  ++out;
131  quotient /= 10;
132  }
133  if (in < 0)
134  *out++ = '-';
135  *out = 0;
136  std::reverse(res,out);
137  return res;
138 }
